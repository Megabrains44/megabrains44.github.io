\begin{solcard}{C10}{Obtain an expression for the magnetic moment $m'$ induced in a steel ball radius $R$ when it is at a distance $x$ from the magnetic dipole $m$.}{0.20}

\par\smallskip\noindent\textbf{Answer:} \textbackslash{}textbf{Answer:} $$m' = \cfrac{2mR^3}{x^3} $$
\end{solcard}
