\title{\textbf{Complex oscillatory system}}
\author{\normalsize Y13 (2013) --- English translation}
\date{}
\begin{document}
\maketitle

The mass $m_1$ is connected by threads to two identical masses $m_2$. The threads are thrown through small light blocks, the distance between which is $L$.

\begin{figure}[H]
    \centering
    \includegraphics[width=0.72\textwidth]{T8_EN_assets/fig1.jpg}
    \caption{Figure 1.}
\end{figure}

\problembox{A1}{At what mass ratio $m_1/m_2$ will the system be in a stable equilibrium position (mass loads $m_2$ do not touch the blocks)?}{0}

\problembox{A2}{Find the period of small oscillations of such a system (the mass $m_1$ oscillates along a vertical straight line).}{0}

\problembox{A3}{Consider the special case of oscillations when $m_1 = m_2$.}{0}
